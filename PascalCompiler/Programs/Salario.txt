program Salario;

var
    salDiario, dias, salarioTot: integer;

begin
    { Initialize rectangle dimensions }
    salDiario := 500;
    dias := 30;

    { Calculate the area of the rectangle }
    salarioTot := dias * salDiario;

    { Display the area }
    writeln("Salario:", salarioTot);

    { Simple if}
    if salarioTot < 11500 then
        writeln("Estas pero pobre");  { mas del salario minimo }
end.
